                 String name = Constants.CURRENT_USER;
                 String date = df.format(new Date());
                 String message = chatArea.getText().trim();
                 chatArea.addChatBox(new ModelMessage(icon, name, date, message), ChatBox.BoxType.RIGHT);
                 chatArea.clearTextAndGrabFocus();

                 name = "ChatBot";
                 chatArea.addChatBox(new ModelMessage(icon, name, date, chatbotDataAccessInterface.execute(message)), ChatBox.BoxType.LEFT);

             }
